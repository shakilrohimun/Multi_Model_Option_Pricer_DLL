70double MonteCarloPricer::price(const Option& opt) const {
71 // Retrieve option parameters.
72 double S0 = opt.getUnderlying(); // Underlying asset price.
73 double K = opt.getStrike(); // Strike price.
74 double sigma = opt.getVolatility(); // Volatility.
75 double q = opt.getDividend(); // Dividend yield.
76
77 // Retrieve simulation parameters from configuration.
78 double T = config_.maturity; // Configured maturity.
79 double r_default = config_.riskFreeRate; // Default risk-free rate.
80 const int NPaths = config_.mcNumPaths; // Number of simulation paths.
81 const int NSteps = config_.mcTimeStepsPerPath; // Number of time steps per path.
82
83 // Adjust effective time to maturity based on the calculation date.
84 double T_effective = T;
85 if (!config_.calculationDate.empty()) {
86 auto calcDate = DateConverter::parseDate(config_.calculationDate);
87 auto today = std::chrono::system_clock::now();
88 double offset = DateConverter::yearsBetween(calcDate, today);
89 if (offset >= T) {
90 throw std::runtime_error("Effective maturity is negative. Check the calculation date.");
91 }
92 T_effective = T - offset;
93 }
94 double dt = T_effective / NSteps;
95
96 // Initialize random number generator with a fixed seed for reproducibility.
97 std::mt19937 rng(42);
98 std::normal_distribution<double> norm(0.0, 1.0);
99
100 if (opt.getOptionStyle() == Option::OptionStyle::European) {
101 // Standard Monte Carlo simulation for European options.
102 double sumPayoff = 0.0;
103 for (int i = 0; i < NPaths; i++) {
104 double S = S0;
105 double disc = 1.0;
106 // Simulate one price path using GBM with variable interest rate.
107 for (int j = 0; j < NSteps; j++) {
108 double t_current = j * dt;
109 double normTime = t_current / T_effective; // Normalized time in [0,1].
110 double r_local = (config_.yieldCurve.getData().empty()) ? r_default : config_.yieldCurve.getRate(normTime);
111 double Z = norm(rng);
112 S *= std::exp((r_local - q - 0.5 * sigma * sigma) * dt + sigma * std::sqrt(dt) * Z);
113 disc *= std::exp(-r_local * dt);
114 }
115 double payoff = (opt.getOptionType() == Option::OptionType::Call) ? std::max(S - K, 0.0)
116 : std::max(K - S, 0.0);
117 sumPayoff += payoff * disc; // Payoff is already discounted along the path.
118 }
119 double priceMC = sumPayoff / NPaths;
120 return priceMC;
121 }
122 else {
123 // Monte Carlo simulation for American options using Longstaff-Schwartz.
124 // Build a grid to store simulated paths.
125 std::vector<std::vector<double>> paths(NPaths, std::vector<double>(NSteps + 1, 0.0));
126 for (int i = 0; i < NPaths; i++) {
127 paths[i][0] = S0;
128 for (int j = 1; j <= NSteps; j++) {
129 double t_current = (j - 1) * dt;
130 double normTime = t_current / T_effective;
131 double r_local = (config_.yieldCurve.getData().empty()) ? r_default : config_.yieldCurve.getRate(normTime);
132 double Z = norm(rng);
133 paths[i][j] = paths[i][j - 1] * std::exp((r_local - q - 0.5 * sigma * sigma) * dt + sigma * std::sqrt(dt) * Z);
134 }
135 }
136 // Initialize cash flows at maturity.
137 std::vector<double> cashFlow(NPaths, 0.0);
138 for (int i = 0; i < NPaths; i++) {
139 cashFlow[i] = (opt.getOptionType() == Option::OptionType::Call) ?
140 std::max(paths[i][NSteps] - K, 0.0) :
141 std::max(K - paths[i][NSteps], 0.0);
142 }
143 // Vector to record exercise time (initially set to maturity).
144 std::vector<int> exerciseTime(NPaths, NSteps);
145
146 // Backward induction using Longstaff-Schwartz.
147 for (int t = NSteps - 1; t >= 1; t--) {
148 std::vector<int> inTheMoneyIndices;
149 std::vector<double> X; // Underlying prices for in-the-money paths.
150 std::vector<double> Y; // Discounted cash flows for in-the-money paths.
151 for (int i = 0; i < NPaths; i++) {
152 double intrinsic = (opt.getOptionType() == Option::OptionType::Call) ?
153 std::max(paths[i][t] - K, 0.0) :
154 std::max(K - paths[i][t], 0.0);
155 if (intrinsic > 0.0 && exerciseTime[i] == NSteps) {
156 inTheMoneyIndices.push_back(i);
157 X.push_back(paths[i][t]);
158 // Compute discount factor from time t to exerciseTime using variable rates.
159 double disc = 1.0;
160 for (int k = t; k < exerciseTime[i]; k++) {
161 double normTime_k = ((T_effective - k * dt) / T_effective);
162 double r_local = (config_.yieldCurve.getData().empty()) ? r_default : config_.yieldCurve.getRate(normTime_k);
163 disc *= std::exp(-r_local * dt);
164 }
165 Y.push_back(cashFlow[i] * disc);
166 }
167 }
168 if (!inTheMoneyIndices.empty()) {
169 int n = inTheMoneyIndices.size();
170 double sum1 = 0.0, sumX = 0.0, sumX2 = 0.0, sumX3 = 0.0, sumX4 = 0.0;
171 double sumY = 0.0, sumXY = 0.0, sumX2Y = 0.0;
172 for (int k = 0; k < n; k++) {
173 double x = X[k];
174 double y = Y[k];
175 sum1 += 1.0;
176 sumX += x;
177 sumX2 += x * x;
178 sumX3 += x * x * x;
179 sumX4 += x * x * x * x;
180 sumY += y;
181 sumXY += x * y;
182 sumX2Y += x * x * y;
183 }
184 double D = sum1 * (sumX2 * sumX4 - sumX3 * sumX3)
185 - sumX * (sumX * sumX4 - sumX2 * sumX3)
186 + sumX2 * (sumX * sumX3 - sumX2 * sumX2);
187 if (std::abs(D) < 1e-10) {
188 continue;
189 }
190 double a0 = (sumY * (sumX2 * sumX4 - sumX3 * sumX3)
191 - sumX * (sumXY * sumX4 - sumX3 * sumX2Y)
192 + sumX2 * (sumXY * sumX3 - sumX2 * sumX2Y)) / D;
193 double a1 = (sum1 * (sumXY * sumX4 - sumX3 * sumX2Y)
194 - sumY * (sumX * sumX4 - sumX2 * sumX3)
195 + sumX2 * (sumX * sumX2Y - sumX2 * sumXY)) / D;
196 double a2 = (sum1 * (sumX2 * sumX2Y - sumX3 * sumXY)
197 - sumX * (sumX * sumX2Y - sumX2 * sumXY)
198 + sumY * (sumX * sumX3 - sumX2 * sumX2)) / D;
199
200 for (int idx = 0; idx < n; idx++) {
201 int i = inTheMoneyIndices[idx];
202 double x = paths[i][t];
203 double continuation = a0 + a1 * x + a2 * x * x;
204 double immediate = (opt.getOptionType() == Option::OptionType::Call) ?
205 std::max(x - K, 0.0) : std::max(K - x, 0.0);
206 if (immediate > continuation) {
207 cashFlow[i] = immediate;
208 exerciseTime[i] = t;
209 }
210 }
211 }
212 }
213 // Final discounting from time 0 to exerciseTime for each path using variable rates.
214 double sumPayoffs = 0.0;
215 for (int i = 0; i < NPaths; i++) {
216 double disc = 1.0;
217 for (int k = 0; k < exerciseTime[i]; k++) {
218 double normTime_k = (T_effective - k * dt) / T_effective;
219 double r_local = (config_.yieldCurve.getData().empty()) ? r_default : config_.yieldCurve.getRate(normTime_k);
220 disc *= std::exp(-r_local * dt);
221 }
222 sumPayoffs += cashFlow[i] * disc;
223 }
224 return sumPayoffs / NPaths;
225 }
226}

237Greeks MonteCarloPricer::computeGreeks(const Option& opt) const {
238 double h = 0.01 * opt.getUnderlying();
239 double volStep = 0.01;
240 double rStep = 0.001;
241 double timeStep = 1.0 / 365.0; // One day
242
243 double basePrice = price(opt);
244
245 // --- Delta ---
246 Option opt_up = opt;
247 Option opt_down = opt;
248 opt_up.setUnderlying(opt.getUnderlying() + h);
249 opt_down.setUnderlying(opt.getUnderlying() - h);
250 double price_up = price(opt_up);
251 double price_down = price(opt_down);
252 double delta = (price_up - price_down) / (2 * h);
253
254 // --- Gamma ---
255 double gamma = (price_up - 2 * basePrice + price_down) / (h * h);
256
257 // --- Vega ---
258 Option opt_vol_up = opt;
259 Option opt_vol_down = opt;
260 opt_vol_up.setVolatility(opt.getVolatility() + volStep);
261 opt_vol_down.setVolatility(opt.getVolatility() - volStep);
262 double vega = (price(opt_vol_up) - price(opt_vol_down)) / (2 * volStep);
263
264 // --- Theta ---
265 // Create a new configuration with a slightly reduced maturity.
266 PricingConfiguration config_T_down = config_;
267 config_T_down.maturity = config_.maturity - timeStep;
268 // Create a new pricer with this configuration.
269 MonteCarloPricer pricer_T_down(config_T_down);
270 double price_T_down = pricer_T_down.price(opt);
271 // Approximate Theta using a backward finite difference.
272 double theta = (price_T_down - basePrice) / (-timeStep);
273
274 // --- Rho ---
275 PricingConfiguration config_r_up = config_;
276 PricingConfiguration config_r_down = config_;
277 if (!config_.yieldCurve.getData().empty()) {
278 YieldCurve yc_up;
279 YieldCurve yc_down;
280 for (const auto& pt : config_.yieldCurve.getData()) {
281 yc_up.addRatePoint(pt.maturity, pt.rate + rStep);
282 yc_down.addRatePoint(pt.maturity, pt.rate - rStep);
283 }
284 config_r_up.yieldCurve = yc_up;
285 config_r_down.yieldCurve = yc_down;
286 }
287 else {
288 config_r_up.riskFreeRate = config_.riskFreeRate + rStep;
289 config_r_down.riskFreeRate = config_.riskFreeRate - rStep;
290 }
291 MonteCarloPricer pricer_r_up(config_r_up);
292 MonteCarloPricer pricer_r_down(config_r_down);
293 double price_r_up = pricer_r_up.price(opt);
294 double price_r_down = pricer_r_down.price(opt);
295 double rho = (price_r_up - price_r_down) / (2 * rStep);
296
297 Greeks greeks;
298 greeks.delta = delta;
299 greeks.gamma = gamma;
300 greeks.vega = vega;
301 greeks.theta = theta;
302 greeks.rho = rho;
303 return greeks;
304}
