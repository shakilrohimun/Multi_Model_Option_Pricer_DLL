22 // European and American options, and leverages a variable risk-free rate by interpolating values from a yield curve.

78double BinomialPricer::price(const Option& opt) const {
79 // Retrieve basic option parameters.
80 double S = opt.getUnderlying();
81 double K = opt.getStrike();
82 double sigma = opt.getVolatility();
83 double q = opt.getDividend();
84 // Retrieve the maturity and the default risk-free rate from the pricing configuration.
85 double T = config_.maturity;
86 double r_const = config_.riskFreeRate; // fallback if yield curve is not loaded
87
88 // Retrieve the number of steps from the configuration.
89 int N = config_.binomialSteps;
90 double dt = T / N;
91
92 // Compute the up (u) and down (d) factors using the CRR model.
93 double u = std::exp(sigma * std::sqrt(dt));
94 double d = 1.0 / u;
95 // Calculate the risk-neutral probability using the constant risk-free rate.
96 double p = (std::exp((r_const - q) * dt) - d) / (u - d);
97 if (p < 0.0 || p > 1.0) {
98 throw std::runtime_error("Invalid risk-neutral probability in the binomial model.");
99 }
100
101 // Create a vector to store the terminal payoffs.
102 std::vector<double> prices(N + 1);
103 for (int j = 0; j <= N; ++j) {
104 double S_j = S * std::pow(u, j) * std::pow(d, N - j);
105 if (opt.getOptionType() == Option::OptionType::Call) {
106 prices[j] = std::max(S_j - K, 0.0);
107 }
108 else { // Put option
109 prices[j] = std::max(K - S_j, 0.0);
110 }
111 }
112
113 // Backward induction through the binomial tree with a variable interest rate.
114 for (int i = N - 1; i >= 0; --i) {
115 // Compute normalized time for the current step (i/N).
116 double t_norm = static_cast<double>(i) / N;
117 // Obtain the local risk-free rate via the yield curve.
118 // If the yield curve is not loaded, it returns the default risk-free rate.
119 double r_local = config_.yieldCurve.getRate(t_norm);
120 // Compute the discount factor using the local rate.
121 double discountFactor = std::exp(-r_local * dt);
122 // Compute the local risk-neutral probability using the local rate.
123 double p_local = (std::exp((r_local - q) * dt) - d) / (u - d);
124
125 for (int j = 0; j <= i; ++j) {
126 // Calculate the expected continuation value at the current node.
127 double continuation = discountFactor * (p_local * prices[j + 1] + (1.0 - p_local) * prices[j]);
128 if (opt.getOptionStyle() == Option::OptionStyle::American) {
129 // For American options, compare the continuation value with the immediate exercise value.
130 double S_i = S * std::pow(u, j) * std::pow(d, i - j);
131 double intrinsic = 0.0;
132 if (opt.getOptionType() == Option::OptionType::Call) {
133 intrinsic = std::max(S_i - K, 0.0);
134 }
135 else {
136 intrinsic = std::max(K - S_i, 0.0);
137 }
138 prices[j] = std::max(continuation, intrinsic);
139 }
140 else {
141 // For European options, only the continuation value is considered.
142 prices[j] = continuation;
143 }
144 }
145 }
146
147 // The price at the root of the tree is the computed option price.
148 return prices[0];
149}

164Greeks BinomialPricer::computeGreeks(const Option& opt) const {
165 double h = 0.01 * opt.getUnderlying();
166
167 // Compute Delta and Gamma by perturbing the underlying asset price.
168 Option opt_up = opt;
169 Option opt_down = opt;
170 opt_up.setUnderlying(opt.getUnderlying() + h);
171 opt_down.setUnderlying(opt.getUnderlying() - h);
172 double price_up = price(opt_up);
173 double price_down = price(opt_down);
174 double basePrice = price(opt);
175 double delta = (price_up - price_down) / (2 * h);
176 double gamma = (price_up - 2 * basePrice + price_down) / (h * h);
177
178 // Compute Vega by perturbing the volatility.
179 double volStep = 0.01;
180 Option opt_vol_up = opt;
181 Option opt_vol_down = opt;
182 opt_vol_up.setVolatility(opt.getVolatility() + volStep);
183 opt_vol_down.setVolatility(opt.getVolatility() - volStep);
184 double vega = (price(opt_vol_up) - price(opt_vol_down)) / (2 * volStep);
185
186 // --- Theta Implementation ---
187 // Estimate Theta by reducing the maturity by one day (converted to years).
188 double dt_small = 1.0 / 365.0; // One day in years
189 PricingConfiguration config_time = config_;
190 config_time.maturity = config_.maturity - dt_small;
191 BinomialPricer pricer_time(config_time);
192 double price_T_down = pricer_time.price(opt);
193 double theta = (basePrice - price_T_down) / dt_small;
194
195 // --- Modified Implementation of Rho ---
196 // Estimate Rho by perturbing the risk-free rate or the yield curve uniformly.
197 double rStep = 0.001;
198 PricingConfiguration config_r_up = config_;
199 PricingConfiguration config_r_down = config_;
200
201 if (!config_.yieldCurve.getData().empty()) {
202 // Shift all rates in the yield curve uniformly by rStep.
203 YieldCurve yc_up;
204 YieldCurve yc_down;
205 for (const auto& pt : config_.yieldCurve.getData()) {
206 yc_up.addRatePoint(pt.maturity, pt.rate + rStep);
207 yc_down.addRatePoint(pt.maturity, pt.rate - rStep);
208 }
209 config_r_up.yieldCurve = yc_up;
210 config_r_down.yieldCurve = yc_down;
211 }
212 else {
213 // If no yield curve data is available, adjust the constant risk-free rate.
214 config_r_up.riskFreeRate = config_.riskFreeRate + rStep;
215 config_r_down.riskFreeRate = config_.riskFreeRate - rStep;
216 }
217
218 BinomialPricer pricer_r_up(config_r_up);
219 BinomialPricer pricer_r_down(config_r_down);
220 double price_r_up = pricer_r_up.price(opt);
221 double price_r_down = pricer_r_down.price(opt);
222 double rho = (price_r_up - price_r_down) / (2 * rStep);
223
224 // Populate the Greeks structure with the computed values.
225 Greeks g;
226 g.delta = delta;
227 g.gamma = gamma;
228 g.vega = vega;
229 g.theta = theta;
230 g.rho = rho;
231 return g;
232}
