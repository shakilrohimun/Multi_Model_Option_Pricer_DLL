68double CrankNicolsonPricer::price(const Option& opt) const {
69 // Option parameters
70 double S0 = opt.getUnderlying(); // Underlying price
71 double K = opt.getStrike(); // Strike price
72 double sigma = opt.getVolatility(); // Volatility
73 double q = opt.getDividend(); // Continuous dividend yield
74
75 // Retrieve maturity and default risk-free rate from configuration.
76 double T = config_.maturity;
77 double r_default = config_.riskFreeRate;
78
79 // Adjust effective time to maturity using calculationDate if provided.
80 double T_effective = T;
81 if (!config_.calculationDate.empty()) {
82 auto calcDate = DateConverter::parseDate(config_.calculationDate);
83 auto today = std::chrono::system_clock::now();
84 double offset = DateConverter::yearsBetween(calcDate, today);
85 T_effective = T - offset;
86 }
87
88 // Retrieve discretization parameters.
89 const int M = config_.crankSpotSteps; // Number of spatial steps
90 const int N = config_.crankTimeSteps; // Number of time steps
91 double Smax = (config_.S_max > 0.0) ? config_.S_max : std::max(3.0 * K, 3.0 * S0);
92 double dS = Smax / M;
93 double dt = T_effective / N; // Use effective time here
94
95 // Build spatial grid.
96 std::vector<double> S(M + 1);
97 for (int j = 0; j <= M; ++j) {
98 S[j] = j * dS;
99 }
100
101 // Terminal condition: payoff at maturity.
102 std::vector<double> V(M + 1, 0.0);
103 if (opt.getOptionType() == Option::OptionType::Call) {
104 for (int j = 0; j <= M; ++j) {
105 V[j] = std::max(S[j] - K, 0.0);
106 }
107 }
108 else { // Put option
109 for (int j = 0; j <= M; ++j) {
110 V[j] = std::max(K - S[j], 0.0);
111 }
112 }
113
114 // Temporary vector for current time step.
115 std::vector<double> newV(M + 1, 0.0);
116 // Vectors for the tridiagonal system.
117 std::vector<double> a(M - 1, 0.0); // Coefficient for V_{j-1}^{n+1}
118 std::vector<double> b(M - 1, 0.0); // Coefficient for V_{j}^{n+1}
119 std::vector<double> c(M - 1, 0.0); // Coefficient for V_{j+1}^{n+1}
120 std::vector<double> d_vec(M - 1, 0.0); // Right-hand side
121
122 // Backward induction loop (n from N-1 to 0)
123 for (int n = N - 1; n >= 0; --n) {
124 double t = n * dt;
125 // Compute normalized time (for yield curve interpolation).
126 // Here, we define normTime such that normTime = 1 at t = 0 (start) and 0 at t = T_effective (maturity)
127 double normTime = (T_effective - t) / T_effective;
128 // Obtain local risk-free rate from yield curve (if available); otherwise, use default.
129 double r_local = (!config_.yieldCurve.getData().empty()) ? config_.yieldCurve.getRate(normTime) : r_default;
130
131 // Boundary conditions at time t.
132 if (opt.getOptionType() == Option::OptionType::Call) {
133 newV[0] = 0.0;
134 newV[M] = Smax - K * std::exp(-r_local * (T_effective - t));
135 }
136 else {
137 newV[0] = K * std::exp(-r_local * (T_effective - t));
138 newV[M] = 0.0;
139 }
140
141 // Form the tridiagonal system for interior nodes j = 1 to M-1.
142 for (int j = 1; j < M; ++j) {
143 double S_j = S[j];
144 // Use local risk-free rate in coefficients.
145 double A = 0.5 * dt * (0.5 * sigma * sigma * S_j * S_j / (dS * dS) - (r_local - q) * S_j / (2 * dS));
146 double B = 1.0 + 0.5 * dt * (sigma * sigma * S_j * S_j / (dS * dS) + r_local);
147 double C = 0.5 * dt * (0.5 * sigma * sigma * S_j * S_j / (dS * dS) + (r_local - q) * S_j / (2 * dS));
148
149 // Explicit part coefficients.
150 double D_coef = A;
151 double E_coef = 1.0 - 0.5 * dt * (sigma * sigma * S_j * S_j / (dS * dS) + r_local);
152 double F = C;
153
154 a[j - 1] = -A;
155 b[j - 1] = B;
156 c[j - 1] = -C;
157 d_vec[j - 1] = D_coef * V[j - 1] + E_coef * V[j] + F * V[j + 1];
158 }
159
160 // Adjust right-hand side for boundary conditions.
161 d_vec[0] -= (-a[0]) * newV[0];
162 d_vec[M - 2] -= (-c[M - 2]) * newV[M];
163
164 // Solve the tridiagonal system using the Thomas algorithm.
165 std::vector<double> c_prime(M - 1, 0.0);
166 std::vector<double> d_prime(M - 1, 0.0);
167
168 c_prime[0] = c[0] / b[0];
169 d_prime[0] = d_vec[0] / b[0];
170
171 for (int j = 1; j < M - 1; ++j) {
172 double m = b[j] - a[j] * c_prime[j - 1];
173 c_prime[j] = c[j] / m;
174 d_prime[j] = (d_vec[j] - a[j] * d_prime[j - 1]) / m;
175 }
176
177 newV[M - 1] = d_prime[M - 2];
178 for (int j = M - 2; j >= 1; --j) {
179 newV[j] = d_prime[j - 1] - c_prime[j - 1] * newV[j + 1];
180 }
181
182 // Update V for the next time step.
183 for (int j = 0; j <= M; ++j) {
184 V[j] = newV[j];
185 }
186
187 // Projection step for American options: update value with intrinsic payoff.
188 if (opt.getOptionStyle() == Option::OptionStyle::American) {
189 for (int j = 0; j <= M; ++j) {
190 double payoff = 0.0;
191 if (opt.getOptionType() == Option::OptionType::Call) {
192 payoff = std::max(S[j] - K, 0.0);
193 }
194 else {
195 payoff = std::max(K - S[j], 0.0);
196 }
197 V[j] = std::max(V[j], payoff);
198 }
199 }
200 }
201
202 // Linear interpolation to obtain the option price at S0.
203 double price = 0.0;
204 if (S0 <= 0) {
205 price = V[0];
206 }
207 else if (S0 >= Smax) {
208 price = V[M];
209 }
210 else {
211 int j = static_cast<int>(S0 / dS);
212 double weight = (S0 - S[j]) / dS;
213 price = V[j] * (1.0 - weight) + V[j + 1] * weight;
214 }
215
216 return price;
217}

228Greeks CrankNicolsonPricer::computeGreeks(const Option& opt) const {
229 double h = 0.01 * opt.getUnderlying();
230 double volStep = 0.01;
231 double rStep = 0.001;
232 double timeStep = 1.0 / 365.0; // One day
233
234 double basePrice = price(opt);
235
236 // --- Delta ---
237 Option opt_up = opt;
238 Option opt_down = opt;
239 opt_up.setUnderlying(opt.getUnderlying() + h);
240 opt_down.setUnderlying(opt.getUnderlying() - h);
241 double price_up = price(opt_up);
242 double price_down = price(opt_down);
243 double delta = (price_up - price_down) / (2 * h);
244
245 // --- Gamma ---
246 double gamma = (price_up - 2 * basePrice + price_down) / (h * h);
247
248 // --- Vega ---
249 Option opt_vol_up = opt;
250 Option opt_vol_down = opt;
251 opt_vol_up.setVolatility(opt.getVolatility() + volStep);
252 opt_vol_down.setVolatility(opt.getVolatility() - volStep);
253 double vega = (price(opt_vol_up) - price(opt_vol_down)) / (2 * volStep);
254
255 // --- Theta ---
256 // Use a time step of one day (in years)
257 double CranktimeStep = 1.0 / 365.0;
258 // Create a new configuration with maturity reduced by timeStep
259 PricingConfiguration config_T_down = config_;
260 config_T_down.maturity = config_.maturity - CranktimeStep;
261 // Recalculate price with the modified configuration
262 CrankNicolsonPricer pricer_T_down(config_T_down);
263 double price_T_down = pricer_T_down.price(opt);
264 // Calculate Theta using a backward finite difference
265 double theta = (price_T_down - basePrice) / (-CranktimeStep);
266
267 // --- Rho ---
268 PricingConfiguration config_r_up = config_;
269 PricingConfiguration config_r_down = config_;
270 if (!config_.yieldCurve.getData().empty()) {
271 YieldCurve yc_up;
272 YieldCurve yc_down;
273 for (const auto& pt : config_.yieldCurve.getData()) {
274 yc_up.addRatePoint(pt.maturity, pt.rate + rStep);
275 yc_down.addRatePoint(pt.maturity, pt.rate - rStep);
276 }
277 config_r_up.yieldCurve = yc_up;
278 config_r_down.yieldCurve = yc_down;
279 }
280 else {
281 config_r_up.riskFreeRate = config_.riskFreeRate + rStep;
282 config_r_down.riskFreeRate = config_.riskFreeRate - rStep;
283 }
284 CrankNicolsonPricer pricer_r_up(config_r_up);
285 CrankNicolsonPricer pricer_r_down(config_r_down);
286 double price_r_up = pricer_r_up.price(opt);
287 double price_r_down = pricer_r_down.price(opt);
288 double rho = (price_r_up - price_r_down) / (2 * rStep);
289
290 Greeks greeks;
291 greeks.delta = delta;
292 greeks.gamma = gamma;
293 greeks.vega = vega;
294 greeks.theta = theta;
295 greeks.rho = rho;
296 return greeks;
297}
