98double BlackScholesPricer::price(const Option& opt) const {
99 if (opt.getOptionStyle() != Option::OptionStyle::European) {
100 throw std::runtime_error("BlackScholesPricer supports only European options.");
101 }
102
103 double S = opt.getUnderlying(); // Underlying price
104 double K = opt.getStrike(); // Strike price
105 double sigma = opt.getVolatility(); // Volatility
106 double q = opt.getDividend(); // Continuous dividend yield
107
108 // Retrieve maturity and default risk-free rate from configuration.
109 double T = config_.maturity;
110 double r_default = config_.riskFreeRate;
111
112 // Adjust effective time to maturity using the calculation date if provided.
113 double T_effective = T;
114 if (!config_.calculationDate.empty()) {
115 auto calcDate = DateConverter::parseDate(config_.calculationDate);
116 auto today = std::chrono::system_clock::now();
117 double offset = DateConverter::yearsBetween(calcDate, today);
118 T_effective = T - offset;
119 }
120
121 // Determine effective risk-free rate using the yield curve if available.
122 double effective_r = r_default;
123
124 double d1 = (std::log(S / K) + (effective_r - q + 0.5 * sigma * sigma) * T_effective) /
125 (sigma * std::sqrt(T_effective));
126 double d2 = d1 - sigma * std::sqrt(T_effective);
127
128 if (opt.getOptionType() == Option::OptionType::Call) {
129 return S * std::exp(-q * T_effective) * norm_cdf(d1) -
130 K * std::exp(-effective_r * T_effective) * norm_cdf(d2);
131 }
132 else {
133 return K * std::exp(-effective_r * T_effective) * norm_cdf(-d2) -
134 S * std::exp(-q * T_effective) * norm_cdf(-d1);
135 }
136}

156Greeks BlackScholesPricer::computeGreeks(const Option& opt) const {
157 if (opt.getOptionStyle() != Option::OptionStyle::European) {
158 throw std::runtime_error("BlackScholesPricer supports only European options.");
159 }
160
161 double S = opt.getUnderlying();
162 double K = opt.getStrike();
163 double sigma = opt.getVolatility();
164 double q = opt.getDividend();
165
166 double T = config_.maturity;
167 double r_default = config_.riskFreeRate;
168
169 // Adjust effective time to maturity using the calculation date if provided.
170 double T_effective = T;
171 if (!config_.calculationDate.empty()) {
172 auto calcDate = DateConverter::parseDate(config_.calculationDate);
173 auto today = std::chrono::system_clock::now();
174 double offset = DateConverter::yearsBetween(calcDate, today);
175 T_effective = T - offset;
176 }
177
178 // Determine effective risk-free rate.
179 double effective_r = r_default;
180
181 double d1 = (std::log(S / K) + (effective_r - q + 0.5 * sigma * sigma) * T_effective) /
182 (sigma * std::sqrt(T_effective));
183 double d2 = d1 - sigma * std::sqrt(T_effective);
184
185 Greeks greeks;
186 if (opt.getOptionType() == Option::OptionType::Call) {
187 greeks.delta = std::exp(-q * T_effective) * norm_cdf(d1);
188 }
189 else {
190 greeks.delta = -std::exp(-q * T_effective) * norm_cdf(-d1);
191 }
192
193 greeks.gamma = std::exp(-q * T_effective) * norm_pdf(d1) /
194 (S * sigma * std::sqrt(T_effective));
195 greeks.vega = S * std::exp(-q * T_effective) * norm_pdf(d1) *
196 std::sqrt(T_effective);
197
198 if (opt.getOptionType() == Option::OptionType::Call) {
199 greeks.theta = -(S * sigma * std::exp(-q * T_effective) * norm_pdf(d1)) /
200 (2 * std::sqrt(T_effective))
201 - effective_r * K * std::exp(-effective_r * T_effective) * norm_cdf(d2)
202 + q * S * std::exp(-q * T_effective) * norm_cdf(d1);
203 }
204 else {
205 greeks.theta = -(S * sigma * std::exp(-q * T_effective) * norm_pdf(d1)) /
206 (2 * std::sqrt(T_effective))
207 + effective_r * K * std::exp(-effective_r * T_effective) * norm_cdf(-d2)
208 - q * S * std::exp(-q * T_effective) * norm_cdf(-d1);
209 }
210
211 if (opt.getOptionType() == Option::OptionType::Call) {
212 greeks.rho = K * T_effective * std::exp(-effective_r * T_effective) * norm_cdf(d2);
213 }
214 else {
215 greeks.rho = -K * T_effective * std::exp(-effective_r * T_effective) * norm_cdf(-d2);
216 }
217
218 // Invert the sign of theta to follow the convention where theta is positive when the option loses value over time.
219 greeks.theta = -greeks.theta;
220
221 return greeks;
